113     /**
114      * Gets the specified <tt>PLAY2_HOME</tt> location.
115      * This method checks in this order:
116      * <ul>
117      * <li>the PLAY2_HOME system variable</li>
118      * <li>the <tt>play2Home</tt> settings</li>
119      * <li>the PLAY2_HOME environment variable</li>
120      * </ul>
121      * If none is set, this method throws an exception.
122      *
123      * @return the play2 location or <code>null</code> if not specified.
124      */
125     public String getPlay2Home() {
126         // First check, system variable
127         String home = System.getProperty(ENV_PLAY2_HOME);
128         if (home != null && !home.isEmpty()) {
129             getLog().debug("Get Play2 home from system variable");
130             return home;
131         }
132 
133         // Second check, the setting configuration
134         if (play2Home != null && !play2Home.isEmpty()) {
135             getLog().debug("Get Play2 home from settings");
136             return play2Home;
137         }
138 
139         // Third check, environment variable
140         home = System.getenv(ENV_PLAY2_HOME);
141         if (home != null && !home.isEmpty()) {
142             getLog().debug("Get Play2 home from environment");
143             return home;
144         }
145 
146         return null;
147     }

149     public File getPlay2() throws MojoExecutionException {
150         // Do we have a cached value ?
151         if (play2executable != null) {
152             return play2executable;
153         }
154 
155         File play2 = null;
156 
157         // Either PLAY2_HOME is defined or not.
158         // In the first case, we're looking for PLAY2_HOME/play[.bat]
159         // In the second case we iterate over the PATH.
160         String path = getPlay2Home();
161         if (path != null) {
162             if (isWindows()) {
163                 play2 = new File(path, "play.bat");
164             } else {
165                 play2 = new File(path, "play");
166             }
167             if (play2.isFile()) {
168                 play2 = manageHomebrew(play2);
169             } else {
170                 throw new MojoExecutionException(ENV_PLAY2_HOME + " system/configuration/environment variable is set " +
171                         "to " + path + " but can't find the 'play' executable");
172             }
173         } else {
174             getLog().info("Looking for 'play' in the System PATH");
175             play2 = findPlay2ExecutableInSystemPath();
176         }
177 
178         if (play2 == null || !play2.isFile()) {
179             throw new MojoExecutionException("Can't find the 'play' executable. Set the " + ENV_PLAY2_HOME + " system/" +
180                     "configuration/environment variable or check the the 'play' executable is available from the " +
181                     "path");
182         }
183 
184         getLog().debug("Using " + play2.getAbsolutePath());
185         play2executable = play2;
186 
187         return play2;
188     }

190     /**
191      * Checks whether the given play executable is in a <tt>Homebrew</tt> managed location.
192      * Homebrew scripts seems to be an issue for play as the provided play executable from this directory is using a
193      *  path expecting relative directories. So, we get such kind of error:
194      *  <br/>
195      * <code>
196      *    /usr/local/Cellar/play/2.0/libexec/play: line 51:
197      *    /usr/local/Cellar/play/2.0/libexec//usr/local/Cellar/play/2.0/libexec/../libexec/framework/build:
198      *    No such file or directory
199      * </code>
200      * <br/>
201      * In this case we substitute the play executable with the one installed by Homebrew but working correctly.
202      * @param play2 the found play2 executable
203      * @return the given play2 executable except if Homebrew is detected, in this case <tt>/usr/local/bin/play</tt>.
204      */
205     private File manageHomebrew(File play2) {
206         if (play2.getAbsolutePath().contains("/Cellar/play/")) {
207             getLog().info("Homebrew installation of play detected");
208             // Substitute the play executable by the homebrew one.
209             File file = new File("/usr/local/bin/play");
210             if (! file.exists()) {
211                 getLog().error("Homebrew installation detected, but no play executable in /usr/local/bin");
212             } else {
213                 return file;
214             }
215         }
216         return play2;
217     }

219     private File findPlay2ExecutableInSystemPath() {
220         String play2 = "play";
221         if (isWindows()) {
222             play2 = "play.bat";
223         }
224         String systemPath = System.getenv("PATH");
225 
226         // Fast failure if we don't have the PATH defined.
227         if (systemPath == null) {
228             return null;
229         }
230 
231         String[] pathDirs = systemPath.split(File.pathSeparator);
232 
233         for (String pathDir : pathDirs) {
234             File file = new File(pathDir, play2);
235             if (file.isFile()) {
236                 return file;
237             }
238         }
239         // Search not successful.
240         return null;
241     }

253     /**
254      * Gets the execution environment.
255      * This method builds a map containing the Maven properties to give to SBT invocations.
256      * It contains maven project data (GAV), pom properties, and system properties.
257      * @return the map of properties (<code><pre>key -> value</pre></code>)
258      */
259     public Map<String, String> getEnvironment() {
260         Map<String, String> env = new HashMap<String, String>();
261         
262         // Environment variables
263         env.putAll( System.getenv() );
264 
265         // Build properties.
266         env.put("project.groupId", project.getGroupId());
267         env.put("project.artifactId", project.getArtifactId());
268         env.put("project.version", project.getVersion());
269 
270         // Pom properties
271         Properties props = project.getProperties();
272         if (props != null) {
273             for (Map.Entry<Object, Object> entry : props.entrySet()) {
274                 env.put(entry.getKey().toString(), entry.getValue().toString());
275             }
276         }
277 
278         // Environment properties
279         Map<String, String> environment = System.getenv();
280         for (String k : environment.keySet()) {
281             env.put(k, environment.get(k));
282         }
283         
284         // System properties
285         props = System.getProperties();
286         if (props != null) {
287             for (Map.Entry<Object, Object> entry : props.entrySet()) {
288                 env.put(entry.getKey().toString(), entry.getValue().toString());
289             }
290         }
291         
292         getLog().debug("Environment built : " + env);
293         return env;
294     }
