30  import static org.mockito.Mockito.*;

32  /**
33   * Test the Packaging Mojo.
34   */
35  public class EnvPropagationMojoTest {
36  
37      @Test
38      public void testPackagingOfMavenAppWithCustomGAV() throws IOException, MojoExecutionException {
39          if (! Helper.detectPlay2()) {
40              System.err.println("PLAY2_HOME missing, skipping tests");
41              return;
42          }
43  
44          File baseDir = new File("target/tests/testPackagingOfMavenAppWithCustomGAV");
45          Helper.copyMavenApp(baseDir);
46  
47          Play2PackageMojo mojo = new Play2PackageMojo();
48  
49          mojo.project = mock(MavenProject.class);
50          mojo.projectHelper = mock(MavenProjectHelper.class);
51          mojo.attachDist = true;
52          mojo.buildDist = true;
53          mojo.deleteDist = false;
54          mojo.setLog(new SystemStreamLog());
55          Build build = mock(Build.class);
56          Artifact artifact = mock(Artifact.class);
57  
58          when(mojo.project.getBasedir()).thenReturn(baseDir);
59          when(mojo.project.getBuild()).thenReturn(build);
60          when(mojo.project.getArtifact()).thenReturn(artifact);
61          when(build.getFinalName()).thenReturn("my-artifact-id-1.0.0");
62          when(mojo.project.getArtifactId()).thenReturn("my-artifact-id");
63          when(mojo.project.getGroupId()).thenReturn("my-group-id");
64          when(mojo.project.getVersion()).thenReturn("1.0.0");
65  
66          mojo.execute();
67  
68          // Verify attached file
69          File target = new File(baseDir, "target");
70          File dist = new File(target, "my-artifact-id-1.0.0.zip");
71          File pack = new File(target, "my-artifact-id-1.0.0.jar");
72          assertThat(dist).exists();
73          assertThat(pack).exists();
74          verify(artifact).setFile(pack);
75          verify(mojo.projectHelper).attachArtifact(mojo.project, "zip", null, dist);
76      }
