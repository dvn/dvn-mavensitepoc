38  /**
39   * Package the Play application.
40   * The application is packaged as a Jar file. It is also possible to create the distribution package (zip).
41   *
42   * @goal package
43   * @phase package
44   */
45  public class Play2PackageMojo
46          extends AbstractPlay2Mojo {
47  
48      /**
49       * Output file classifier.
50       *
51       * @parameter default-value=""
52       */
53      String classifier;
54  
55      /**
56       * Enables or disables the packaging of the whole distribution. The distribution is an autonomous archive containing
57       * all the files required to run the play application. Play 2 module can disable the distribution packaging.
58       *
59       * @parameter default-value=true
60       */
61      boolean buildDist;
62  
63      /**
64       * Enables or disables the attachment of the distribution file as an artifact to this project.
65       * This option has no impact if the distribution is not built.
66       *
67       * @parameter default-value=true
68       */
69      boolean attachDist;
70  
71      /**
72       * Enables or disables the deletion of the <tt>dist</tt> folder after having packaged the application and copied the
73       * distribution file to <tt>target</tt>. It allows keeping the base directory cleaner.
74       *
75       * @parameter default-value=true
76       */
77      boolean deleteDist;
78  
79      /**
80       * Allows customization of the play packaging. The files specified in this attribute will get added to the distribution
81       * zip file. This allows, for example, to write your own start script and have it packaged in the distribution.
82       * This is done post-packaging by the play framework.
83       *
84       * @parameter
85       */
86      List<String> additionalFiles = new ArrayList<String>();
87  
88      public void execute()
89              throws MojoExecutionException {
90  
91          // Package
92          packageApplication();
93          File packagedApplication = moveApplicationPackageToTarget();
94  
95          // Distribution
96          File dist = null;
97          if (buildDist) {
98              packageDistribution();
99              dist = moveDistributionArtifactToTarget();
100 
101             if (!additionalFiles.isEmpty()) {
102                 packageAdditionalFiles(additionalFiles, dist);
103             }
104         }
105         attachArtifactsToProject(packagedApplication, dist);
106     }

108     private void packageAdditionalFiles(List<String> additionalFiles, File distributionFile) throws MojoExecutionException {
109         try {
110             ZipFile zipFile = new ZipFile(distributionFile);
111 
112             ArrayList<File> filesToAdd = new ArrayList<File>(additionalFiles.size());
113             for (String file : additionalFiles) {
114                 File fileToAdd = new File(file);
115                 if (!fileToAdd.exists()) {
116                     throw new MojoExecutionException(fileToAdd.getCanonicalPath() + " not found, can't add to package");
117                 }
118                 filesToAdd.add(fileToAdd);
119 
120                 ZipParameters parameters = new ZipParameters();
121                 parameters.setCompressionLevel(Zip4jConstants.DEFLATE_LEVEL_FASTEST);
122                 parameters.setIncludeRootFolder(true);
123                 // Let's safely assume that the zip filename is also the root directory all files are packaged in
124                 parameters.setRootFolderInZip(StringUtils.substringBeforeLast(distributionFile.getName(), ".zip"));
125                 parameters.setReadHiddenFiles(true);
126 
127                 String message = String.format("Adding files to distribution zip [%s]: \n\t%s",
128                         distributionFile.getCanonicalPath(), StringUtils.join(additionalFiles, "\n\t"));
129                 getLog().info(message);
130 
131                 zipFile.addFiles(filesToAdd, parameters);
132             }
133         } catch (ZipException e) {
134             throw new MojoExecutionException("Cannot add files to zipfile: " + distributionFile, e);
135         } catch (IOException e) {
136             throw new MojoExecutionException("Cannot add files to zipfile: " + distributionFile, e);
137         }
138     }

140     private File moveApplicationPackageToTarget() throws MojoExecutionException {
141         File target = new File(project.getBasedir(), "target");
142         File[] files = FileUtils.convertFileCollectionToFileArray(
143                 FileUtils.listFiles(target, new PackageFileFilter(), new PrefixFileFilter("scala-")));
144         if (files.length == 0) {
145             throw new MojoExecutionException("Cannot find packaged file");
146         } else if (files.length > 1) {
147             getLog().error("Cannot find the packaged file - Too many matches");
148             for (File f : files) {
149                 getLog().error("\t " + f.getAbsolutePath());
150             }
151             throw new MojoExecutionException("Cannot find packaged file : " + files.length + " matches");
152         }
153 
154         try {
155             if (StringUtils.isBlank(classifier)) {
156                 File out = new File(target, project.getBuild().getFinalName() + ".jar");
157                 getLog().info("Copying " + files[0].getName() + " to " + out.getName());
158                 FileUtils.copyFile(files[0], out);
159                 return out;
160             } else {
161                 File out = new File(target, project.getBuild().getFinalName() + "-" + classifier + ".jar");
162                 getLog().info("Copying " + files[0].getName() + " to " + out.getName());
163                 FileUtils.copyFile(files[0], out);
164                 return out;
165             }
166         } catch (IOException e) {
167             throw new MojoExecutionException("Cannot copy package file " + files[0].getAbsolutePath()
168                     + " to " + target.getAbsolutePath());
169         }
170     }

172     private void packageApplication() throws MojoExecutionException {
173         String line = getPlay2().getAbsolutePath();
174 
175         CommandLine cmdLine = CommandLine.parse(line);
176         cmdLine.addArgument("package");
177         DefaultExecutor executor = new DefaultExecutor();
178 
179         if (timeout > 0) {
180             ExecuteWatchdog watchdog = new ExecuteWatchdog(timeout);
181             executor.setWatchdog(watchdog);
182         }
183 
184         executor.setWorkingDirectory(project.getBasedir());
185         executor.setExitValue(0);
186         try {
187             executor.execute(cmdLine, getEnvironment());
188         } catch (IOException e) {
189             throw new MojoExecutionException("Error during packaging", e);
190         }
191     }

215     private File moveDistributionArtifactToTarget() throws MojoExecutionException {
216         // The artifact is in dist.
217         File dist = new File(project.getBasedir(), "dist");
218         if (!dist.exists()) {
219             throw new MojoExecutionException("Can't find the 'dist' directory");
220         }
221 
222         Collection<File> found = FileUtils.listFiles(dist, new String[]{"zip"}, false);
223         if (found.size() == 0) {
224             throw new MojoExecutionException("The distribution file was not found in " + dist.getAbsolutePath());
225         } else if (found.size() > 1) {
226             throw new MojoExecutionException("Too many match for the distribution file in " + dist.getAbsolutePath());
227         }
228 
229         // 1 file
230         File file = found.toArray(new File[0])[0];
231 
232         getLog().info("Distribution file found : " + file.getAbsolutePath());
233 
234         File target = new File(project.getBasedir(), "target");
235         File out = null;
236         if (classifier == null) {
237             out = new File(target, project.getBuild().getFinalName() + ".zip");
238         } else {
239             out = new File(target, project.getBuild().getFinalName() + "-" + classifier + ".zip");
240         }
241 
242         try {
243             getLog().info("Copying " + file.getName() + " to " + out.getName());
244             FileUtils.copyFile(file, out, true);
245         } catch (IOException e) {
246             throw new MojoExecutionException("Can't copy the distribution file to the target folder", e);
247         }
248 
249         // Delete the dist folder if enabled.
250         if (deleteDist) {
251             getLog().debug("Deleting " + dist.getAbsolutePath());
252             FileUtils.deleteQuietly(dist);
253         }
254 
255         return out;
256     }

258     private void attachArtifactsToProject(File app, File dist) {
259         Artifact artifact = project.getArtifact();
260 
261         if (StringUtils.isBlank(classifier)) {
262             artifact.setFile(app);
263         } else {
264             projectHelper.attachArtifact(project, "jar", classifier, app);
265         }
266 
267         if (buildDist && attachDist) {
268             projectHelper.attachArtifact(project, "zip", classifier, dist);
269         }
270 
271     }

273     private class PackageFileFilter extends AbstractFileFilter {

275         public boolean accept(File dir, String name) {
276             // Must be in the scala dir
277             return dir.getName().startsWith("scala")
278                     // Must ends with Jar
279                     && name.endsWith(".jar");
